Gradle Exapmle 2 (war)



==== build.gradle ====
plugins {
id 'war' // Будет создан war файл
}

repositories { // Центральный репозиторий
mavenCentral()
}

dependencies { // Зависимости будут выкачиваться из репозитроия (repositories)
implementation 'org.apache.tomcat:tomcat-servlet-api:10.0.14'
implementation 'org.apache.tomcat:tomcat-jsp-api:10.0.14'
}
