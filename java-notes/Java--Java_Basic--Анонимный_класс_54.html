Анонимный класс


public static void main(String[] args) {
        System.out.println("Hello Java!");
        /** Анонимный класс можно использовать,
         * если немного функионала(логики).
         * У экземпляра "rect" нет имени класса.
         * У этого экземпляра создается новая реализация
         * сразу же с созданием экземпляра
         */
        Rectifier rect = new Rectifier() { //Набрать: Rectifier r = new[пробел]ввод
            int x;//Если необходимо хранить состояние,
            // то проинициализировать это можно с помощью инициализатора
            { //Инициализатор "{}" вызовется перед тем, как вызовется конструктор
                x = 1258;
            }
            static { //Статический инициализатор нужен,
                // чтобы проинициализировать статические переменные
            }

            @Override
            public void rect() {
            }

            @Override
            public int placerect() {
                return 0;
            }
        };
}
