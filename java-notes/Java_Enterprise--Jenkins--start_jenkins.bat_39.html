start_jenkins.bat


start_jenkins.bat - для запуска Jenkins, параметры JAVA_HOME и PATH подставит только на время запуска

set JAVA_HOME=C:\Java\openjdk-17.0.2\bin

set PATH=C:\Java\openjdk-17.0.2\bin

java -jar jenkins.war --httpPort=1222

или

set JAVA_HOME=C:\Java\openjdk-17.0.2

set PATH=%JAVA_HOME%\bin;%PATH%

echo JAVA_HOME=%JAVA_HOME%

echo PATH=%PATH%

java -version

###echo PATH=%PATH%

java -jar jenkins.war --httpPort=1222
