++ (префиксный инкремент)int a = 8;

int b = ++a;

System.out.println(a); // 9

System.out.println(b); // 9

++ (постфиксный инкремент)int a = 8;

int b = a++;

System.out.println(a); // 9

System.out.println(b); // 8

-- (префиксный декремент)int b = --a;
