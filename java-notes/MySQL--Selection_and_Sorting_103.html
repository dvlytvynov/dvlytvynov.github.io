Selection and Sorting


Выборка по условию

В окне запросов вводим

USE employees; Ctrl + Enter // Выбираем базу данных employees

SELECT * FROM employees; // Получаем всех пользователей из таблицы employees

• SELECT - говорит о том, что будем извлекать данные

• * - хотим видеть все столбцы в точно таком же порядке, в каком они определены в таблице

• FROM employees - информацию извлекаем из таблицы employees

SELECT first_name, last_name, gender FROM employees; // Получаем информацию о всех пользователях только по столбцам first_name, last_name, gender

SELECT first_name, last_name, gender FROM employees WHERE hire_date = '1986-06-26';

• WHERE - после WHERE помещаем условия - критерии выборки, можно применять =, !=, >, <, >=, <=

DESCRIBE employees; // Свойства столбцов таблицы employees

Выборка по нескольким условиям

SELECT * FROM employees WHERE hire_date != '1986-06-26' AND gender = 'F';

SELECT * FROM employees WHERE first_name = 'Sashi' OR first_name = 'Petter' OR first_name = 'Hugo';

Приоритет у AND выше, чем у OR

SELECT * FROM employees WHERE (first_name = 'Sashi' OR first_name = 'Petter' OR first_name = 'Hugo') AND hire_date > '1988-01-01';

Оператор IN работает как набор сравнений, представленных через OR

Результат одинаков:

• SELECT * FROM employees WHERE first_name IN ('Sashi', 'Petter', 'Hugo') AND hire_date > '1988-01-01';

• SELECT * FROM employees WHERE (first_name = 'Sashi' OR first_name = 'Petter' OR first_name = 'Hugo') AND hire_date > '1988-01-01';

Результат одинаков:

• SELECT * FROM employees WHERE hire_date >= '1988-01-01' AND hire_date <= '1988-12-31';

• SELECT * FROM employees WHERE hire_date BETWEEN '1988-01-01' AND '1988-12-31';

Результат одинаков:

• SELECT * FROM film WHERE rating = 'PG-13' AND (rental_rate = 0.99 OR rental_rate = 2.99);

• SELECT * FROM film WHERE rental_rate IN (0.99, 2.99) AND rating = 'PG-13';

Сортировка (ORDER BY)

SELECT first_name, last_name, hire_date FROM employees WHERE hire_date BETWEEN '1988-01-01' AND '1988-12-31' ORDER BY hire_date;

SELECT first_name, last_name, hire_date FROM employees WHERE hire_date BETWEEN '1988-01-01' AND '1988-12-31' ORDER BY first_name DESC; // В обратном проядке

SELECT first_name, last_name, hire_date FROM employees WHERE hire_date BETWEEN '1988-01-01' AND '1988-12-31' ORDER BY first_name DESC, hire_date; // Сначала сортировка по first_name в обратном порядке, а в пределах одинаковых first_name сортировка по hire_date по возрастанию

Ограничения выборки

SELECT * FROM employees ORDER BY hire_date DESC LIMIT 10; // Сначала сортировка по hire_date от большего к меньшему, потом вывод только 10 первых строк

Смещение ограничения выборки (можно использовать для постраничного вывода информации)

SELECT * FROM employees ORDER BY hire_date DESC LIMIT 2, 10; // 2 строки пропустить, следующие 10 строк вывести
