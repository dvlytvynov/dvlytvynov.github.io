<?xml version="1.0" encoding="UTF-8"?>
<project xmlns="http://maven.apache.org/POM/4.0.0"
xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
xsi:schemaLocation="http://maven.apache.org/POM/4.0.0 http://maven.apache.org/xsd/maven-4.0.0.xsd">
<modelVersion>4.0.0</modelVersion>

<groupId>homework.multimodule.project</groupId>
<artifactId>hw24</artifactId>
<version>1.0.0.0-SNAPSHOT</version>

<modules>
<module>interface</module>
<module>realization1</module>
<module>realization2</module>
<module>application</module>
</modules>

<packaging>pom</packaging>

<properties>
<maven.compiler.source>17</maven.compiler.source>
<maven.compiler.target>17</maven.compiler.target>
</properties>

<profiles>
<profile>
<id>dev</id>
<properties>
<!-- Настройки для профиля dev -->
<maven.compiler.source>17</maven.compiler.source>
<maven.compiler.target>8</maven.compiler.target>
</properties>
<dependencyManagement>
<dependencies>
<!-- Эти зависимости будут включены в проект
только тогда, когда мы собираем проект
с включенным профилем dev -->
</dependencies>
</dependencyManagement>
<build>
<pluginManagement>
<plugins>
<!-- Настройки плагинов для профиля dev -->
</plugins>
</pluginManagement>
</build>
</profile>
</profiles>

<dependencyManagement>
<dependencies>
<dependency>
<!-- https://mvnrepository.com/artifact/org.apache.poi/poi -->
<groupId>org.apache.poi</groupId>
<artifactId>poi</artifactId>
<version>5.2.2</version>
</dependency>
<dependency>
<!-- https://mvnrepository.com/artifact/org.apache.commons/commons-csv -->
<groupId>org.apache.commons</groupId>
<artifactId>commons-csv</artifactId>
<version>1.9.0</version>
</dependency>
<dependency>
<!-- https://mvnrepository.com/artifact/org.apache.commons/commons-lang3 -->
<groupId>org.apache.commons</groupId>
<artifactId>commons-lang3</artifactId>
<version>3.12.0</version>
</dependency>
</dependencies>
</dependencyManagement>

<dependencies>
<!-- Добавить те зависимости из dependencyManagement,
которые должны быть доступны во всех модулях -->
</dependencies>

<build>
<pluginManagement>
<!-- pluginManagement работает аналогично dependencyManagement.
Позволяет сконфигурировать в одном месте плагин,
и этот плагин будет выполняться с такими же конфигурациями
во всех дочерних модулях -->
<plugins>

</plugins>
</pluginManagement>
<plugins>
<!-- Можем добавлять плагины; можем написать свой плагин,
который будет выполнять в определенную фазу сборки -->
<plugin>
<groupId>org.apache.maven.plugins</groupId>
<artifactId>maven-compiler-plugin</artifactId>
<!-- Будет использован этот плагин
не с версией по умолчанию,
а с версией, которую укажем -->
<version>3.1</version>
<configuration>
<!-- Сконфигурировать этот плагин -->
</configuration>
</plugin>
</plugins>
</build>

</project>
